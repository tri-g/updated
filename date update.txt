TRANS_LATITUDE FLOAT,
NAME VARCHAR(40),
TRANS_ZIP FLOAT,
TRANS_LONGITUDE FLOAT,
MODE VARCHAR(25),
TRANS_ADDRESS VARCHAR(255),
